# Print out variables (-x is xml)
* nvram -xp
* nvram -p
* nvram variablename
* e.g., 'nvram SystemAudioVolume' or 'nvram boot-args'

# As of 2017 and newer Macs apple has shutoff the startup chime (donâ€™t think it can be enabled)

# Clear all nvram from cli
* nvram -c

# Delete specific variables
* nvram -d (variable key)
* e.g., nvram -d SystemAudioVolume

# Disable mac boot chime from cli
* nvram SystemAudioVolume=%80

# Enable mac boot chime from cli
* nvram -d SystemAudioVolume

# Always boot with verbose
* nvram boot-args="-v"

# Disable boot with verbose
* nvram boot-args=

# Enable safe boot mode
* nvram boot-args="-x"

# Disable safe boot mode
* nvram boot-args=""

# Reset through keys
* Reboot Mac
* Immediately Hold Command-Option-P-R
