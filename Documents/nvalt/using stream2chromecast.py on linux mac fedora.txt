# Describes usage mainly on Fedora

# First fix the firewall on fedora - note we are opening up 8765 for the actual stream

        sudo firewall-cmd --permanent --add-port=32768-61000/udp
        sudo firewall-cmd —permanent —add-port=8765/tcp
        sudo firewall-cmd —reload

# Open a terminal window (keep the window open until movie ends or use nohup)

        nohup ./stream2chromecast.py -devicename Johnnie -port 8765 /path/to/movie &
        
# With nohup you can close the terminal window or reuse it for more commands

# Controlling commands (You can also use your mobile to pause, etc)

        ./stream2chromecast.py -devicename Johnnie [-stop, -pause, -continue, -mute, -volup, -voldown]

# Play from URL

        nohup ./stream2chromecast.py -devicename Johnnie -port 8765 -playurl URL &
        
# Other commands

        **list all devices on network:** ./stream2chromecast.py -devicelist 
        
        **status of device:** ./stream2chromecast.py -devicename Johnnie -status
