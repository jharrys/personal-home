1. how to get search params from route.(js|ts) route handler - https://....?name=harris
export async function GET(request) {
...
const name = request.nextUrl.searchParams.get("name")
// name = harris
}

2. how to get "Dynamic Route Segments" (slugs) from route.(js|ts)
// this is a general call
export async function GET(request, { params })

// you can embellish the params so if you have a slug so your route is /app/api/cdol/[slug]/route.js
export async function GET(request, { params: { slug: slug }}) {
...
// the variables is already defined in the arguments - you'll have slug=SomeValue from the route
}
