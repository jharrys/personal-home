# Create a new repository on the IHC bitbucket server

        curl -X POST -u lpjharri https://git.co.ihc.com/rest/api/1.0/projects/JAH/repos -H "Content-Type: application/json" -d '{"name": “NewRepoName”, "scmId": "git", "forkable": "true"}'
    
# Delete a new repository on the IHC bitbucket server

        curl -X DELETE -u lpjharri https://git.co.ihc.com/rest/api/1.0/projects/JAH/repos/blah
